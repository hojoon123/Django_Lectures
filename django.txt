def variable_view(request):
    
    my_var = {'first_name':'K', 'last_name' : 'HJ',
              'some_list' : [1,2,3], 'some_dic' : {'inside_key' :'inside_value'},
    }
    return render(request,'first_app/variable.html', context= my_var)
